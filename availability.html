  <script>

    const bookedDates = [

      // Add booked dates here in YYYY-MM-DD format, e.g. "2025-08-15"

    ];


    const calendarEl = document.getElementById('calendar');

    const bookingFormContainer = document.getElementById('booking-form-container');

    const bookingForm = document.getElementById('booking-form');

    const dateInput = document.getElementById('date');

    const cancelBtn = document.getElementById('cancel-btn');


    function createCalendar(year, month) {

      calendarEl.innerHTML = '';


      const date = new Date(year, month, 1);

      const firstDay = date.getDay();

      const daysInMonth = new Date(year, month + 1, 0).getDate();


      const table = document.createElement('table');

      table.className = 'calendar-table';


      // Table header with days of week

      const thead = document.createElement('thead');

      const headerRow = document.createElement('tr');

      ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'].forEach(day => {

        const th = document.createElement('th');

        th.textContent = day;

        headerRow.appendChild(th);

      });

      thead.appendChild(headerRow);

      table.appendChild(thead);


      const tbody = document.createElement('tbody');

      let row = document.createElement('tr');


      // Empty cells before first day

      for (let i = 0; i < firstDay; i++) {

        row.appendChild(document.createElement('td'));

      }


      for (let dayNum = 1; dayNum <= daysInMonth; dayNum++) {

        const cell = document.createElement('td');

        const dateStr = `${year}-${String(month + 1).padStart(2,'0')}-${String(dayNum).padStart(2,'0')}`;


        const cellDate = new Date(year, month, dayNum);

        const today = new Date();

        today.setHours(0,0,0,0);


        cell.textContent = dayNum;


        // Disable past dates

        if (cellDate < today) {

          cell.classList.add('past-date');

        }

        // Mark booked dates

        else if (bookedDates.includes(dateStr)) {

          cell.classList.add('booked-date');

          cell.title = 'Booked';

        } 

        // Available dates clickable

        else {

          cell.classList.add('available-date');

          cell.tabIndex = 0;

          cell.style.cursor = 'pointer';

          cell.addEventListener('click', () => openBookingForm(dateStr));

          cell.addEventListener('keypress', e => {

            if (e.key === 'Enter' || e.key === ' ') {

              openBookingForm(dateStr);

            }

          });

        }


        row.appendChild(cell);


        if ((firstDay + dayNum) % 7 === 0) {

          tbody.appendChild(row);

          row = document.createElement('tr');

        }

      }


      // Fill remaining cells in last row

      while (row.children.length < 7) {

        row.appendChild(document.createElement('td'));

      }

      tbody.appendChild(row);


      table.appendChild(tbody);

      calendarEl.appendChild(table);

    }


    function openBookingForm(dateStr) {

      bookingFormContainer.classList.remove('hidden');

      dateInput.value = dateStr;

      window.scrollTo({ top: bookingFormContainer.offsetTop, behavior: 'smooth' });

    }


    cancelBtn.addEventListener('click', () => {

      bookingFormContainer.classList.add('hidden');

      bookingForm.reset();

    });


    // Initialize calendar for current month

    const now = new Date();

    createCalendar(now.getFullYear(), now.getMonth());

  </script>
